使用filebeat收集系统日志

修改filebeat.yml配置文件

[root@efk ~]# cat /etc/filebeat/filebeat.yml 
filebeat.inputs:
- type: log
  enabled: true
  paths:
    - /var/log/messages
      fields:
        index: messages

setup.kibana:
  host: "10.0.0.7:5601"

output.elasticsearch:
  hosts: ["10.0.0.7:9200"]
  username: "elastic"
  password: "123456"
  indices:
    - index: "efk-messages-%{+yyyy.MM.dd}"
      when.equals:
        fields:
          index: "messages"

重启filebeat服务

[root@efk ~]# systemctl restart filebeat

在kibana配置web展示

可以进行 fields 区域筛选，点击 "Available Fields"中的 "message"，然后点击添加 "add"，可以看到按照 "message" 筛选后的结果

filebeat语法介绍

filebeat.inputs:
- type: log
  enabled: true
  paths:
    - /var/log/nginx/access.log
      fields:
        index: nginx
      json.keys_under_root: true
      json.overwrite_keys: true

- type: log
  enabled: true
  paths:
    - /var/log/messages
      fields:
        index: messages

output.elasticsearch:
  hosts: ["10.0.0.51:9200"]
  indices:
    - index: "efk-nginx-%{+yyyy.MM.dd}"
      when.equals:
        fields:
          index: "nginx"
    - index: "efk-messages-%{+yyyy.MM.dd}"
      when.equals:
        fields:
          index: "messages"

filebeat.inputs:					//收集什么日志
- type: log						    //采集类型，如果有多个应用系统则配置多个
  enabled: true						//是否启用，true为启用，false为不起用，启用后就会在es上创建索引库
  paths:						   //日志所在路径
    - /var/log/nginx/access.log	
    - /var/log/nginx/error.log
      fields:						   //在es库里额外增加一个字段，也类似于标签
        index: nginx				   //字段名:值
      json.keys_under_root: true		//解析json格式的日志
      json.overwrite_keys: true

output.elasticsearch:			   //日志存放在哪个es集群
  hosts: ["192.168.81.210:9200"]	//集群地址
  indices:						  //索引信息
    - index: "elk-nginx-%{+yyyy.MM.dd}"	  //索引名称
      when.equals:						//匹配哪个标签
        fields:							//配置标签类型为字段
          index: "nginx"				 //配置指定的字段，当index字段的值为nginx则存储到elk-nginx索引中

不同主机配置filebeat

filebeat.inputs:
- type: log

  hosts: ["10.0.0.51:9200"]
  indices:
  - index: "efk-52-message-%{+yyyy.MM.dd}"
    when.equals:

        index: "52-message"

##### Input Config
filebeat.inputs:
- type: log
  enabled: true
  paths:
    - /data/logs/*.log
  fields:
    log_type: shop-seo
  tail_files: true
  exclude_files: ['.gz$']

fields_under_root: true

filebeat.config.modules:
  path: ${path.config}/modules.d/*.yml
  reload.enabled: false

##### Template Config #####
setup.template.settings:
  index.number_of_shards: 1
setup.template.enabled: false
setup.template.overwrite: true
#可以使用变量或者自行定义

setup.template.pattern: "vk-*"
